class A {
  method() {
    print "A method";
  }
}

class B < A {
  method() {
    print "B method";
  }
  function() {
    this.method();
  }

  test() {
    super.method();
  }
}

class C < B {}

C().test();
B().function();
fun fib(n)
{
    if (n < 2)
        return n;
    return fib(n - 1) + fib(n - 2);
}
print fib(10);
